Q1. Print Output

        function counter(){
            var counter = 0;
        
            function IncreaseCounter() {
                return counter += 1;
            };
        
            return IncreaseCounter;
        }        
        var counter = counter();
        alert(counter());
        alert(counter());
        alert(counter());
        alert(counter());
Ans. This code will produce 1, 2, 3, 4 as alerts respectively.
