                add++;
              }

Q8. Palindrome Check

        Ans. var Palin_Check = (str) =>
                {
                let i = 0;
                let j = str.length -1;
                
                while(i < j){
                if(str[i] === str[j]){
                    i++;
                    j--;
                }
                else{
                    return "False";
                }
                }
                return "True";
                }
